ALGORITHM Sum_Distinct

VAR
    T1 : ARRAY_OF INTEGER[10];
    T2 : ARRAY_OF INTEGER[10];
    n1, n2 : INTEGER;
    i, j : INTEGER;
    existe : BOOLEAN;
    somme : INTEGER;

BEGIN

    Read(n1)

    FOR i FROM 0 TO n1-1 DO
        Read(T1[i])
    END_FOR

    Read(n2)

    FOR i FROM 0 TO n2-1 DO
        Read(T2[i])
    END_FOR

    somme <- 0

    FOR i FROM 0 TO n1-1 DO

        existe <- FALSE

        FOR j FROM 0 TO n2-1 DO

            IF T1[i] = T2[j] THEN
                existe <- TRUE
            END_IF

        END_FOR

        IF existe = FALSE THEN
            somme <- somme + T1[i]
        END_IF

    END_FOR

    FOR i FROM 0 TO n2-1 DO

        existe <- FALSE

        FOR j FROM 0 TO n1-1 DO

            IF T2[i] = T1[j] THEN
                existe <- TRUE
            END_IF

        END_FOR

        IF existe = FALSE THEN
            somme <- somme + T2[i]
        END_IF

    END_FOR

    Write(somme)

END
